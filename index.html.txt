    <input type="url" id="link" placeholder="URL do link" required>
    <input type="url" id="imagem" placeholder="URL da imagem/ícone" required>
    <button onclick="adicionarLink()">Adicionar</button>
  </div>

  <!-- Onde os botões aparecerão -->
  <div id="links"></div>

  <script>
    function adicionarLink() {
      const titulo = document.getElementById("titulo").value;
      const link = document.getElementById("link").value;
      const imagem = document.getElementById("imagem").value;

      if (!titulo || !link || !imagem) {
        alert("Preencha todos os campos!");
        return;
      }

      const container = document.getElementById("links");
      const a = document.createElement("a");
      a.href = link;
      a.target = "_blank";
      a.className = "btn";
      a.innerHTML = `<img src="${imagem}" alt=""> ${titulo}`;
      container.appendChild(a);

      // Limpar os campos
      document.getElementById("titulo").value = "";
      document.getElementById("link").value = "";
      document.getElementById("imagem").value = "";
    }
  </script>
